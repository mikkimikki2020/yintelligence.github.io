Argument checking

class Character(object):  
    
    # The initializer of the instance
    def __init__(self, name="Unknown", age="0"):
        self.name = name
        self.age = age
        self.friends = []
        
    def get_friends_names(self):
        s = ""
        for f in self.friends:
            friend = f.name if f is not self else "themself"
            if f == self.friends[0]:
                s += f"{friend}"
            elif f != self.friends[-1]:
                s += f", {friend}"
            else:
                s += f", and {friend}"
        return s
        
    def get_info_string(self):
        line1 = f"{self.name} is a {type(self).__name__} who is {self.age} years old\n"
        line2 = "They have no friends" if len(self.friends) == 0 else f"They are friends with {self.get_friends_names()}"
        return line1 + line2
        
    def about(self):
        info = self.get_info_string()
        print(info)
        
    def __str__(self):
        info = Character.get_info_string(self)
        return info
    
    def add_friend(self, friend):
        if isinstance(friend, Character) and friend not in self.friends:
            self.friends.append(friend)
        elif isinstance(friend, list) and isinstance(friend[0], Character):
            _ = [self.add_friend(f) for f in friend]
        else:
            raise Warning(f"{friend} is not a character, nor does it inherit from Character")

Overriding inherited methods

class Knight(Character):
    def __init__(self, strength=0, weapon="none", name="Unknown", age=0):
        super().__init__(name, age)
        self.strength = strength
        self.weapon = weapon
        
    def get_info_string(self):
        info = super().get_info_string()
        info += f"\nTheir weapon is their trusty {self.weapon}, and their strength is {self.strength}"
        return info
    
    def native_stats(self):
        kvp = vars(self)
        for key in kvp:
            print(f"{key}: {kvp[key]}")

# The music colleciton
class Library(object):
    def __init__(self):
        self.albums = []
        
    def add_album(self, album):
        if isinstance(album, Library.Album):
            self.albums.append(album)
            return album
        else:
            raise ValueError("Check arument types!")
    
    class Song():
        def __init__(self, artist, name, duration):
            self.artist = artist
            self.name = name
            self.duration = duration
            
    class Album():
        def __init__(self, title):
            self.songs = []
            self.title = title
            
        def add_song(self, song):
            self.songs.append(song)
    
    def get_songs_by_artist(self, artist):
        results = [(song.name, album.title) for album in self.albums for song in album.songs if song.artist == artist]
        return results

Exercise

  1. In order to organize our digital music library, we rely on our implementation in Python.
    1. A song is a music item with an artist, a title and a duration. Make a class to represent a song,with a constructor taking as arguments the artist (a string), its title (will be a string) and the duration (will be a number, for instance in seconds).
    2. An album is a sequence of songs. The album also has a title. Make an album class, with a constructor takes the title as an argument. Add a method ”addSong(song)” that takes a song as an argument and adds it to the end of the album.
    3. Add a method that retrieves all the songs of a given artist in the library. This method should return a list of tuples. The first element in the tuple will be the song, the second element will be the album the song belongs to. On what class should this method be implemented?
  2. In this question, we will build a super small tool for hospitals for managing appointments for their patients.
    1. A patient is an object with a name and a date of birth. Make a patient class, with a constructor taking the name and the date of birth (choose a way to represent dates). Add the correct method to the class such that when a patient object is printed, it shows: “patient //name// born //date of birth//”
    2. Similarly, a doctor is an object with a name and a n associated department. Make a class with a constructor taking these two pieces of information as arguments.
    3. An appointment is an association between a date, a time, a patient, a doctor and a duration. Make a class for the appointment.
    4. An agenda is a collection of appointments. Make a class for the agenda. Add a method to store an appointment for a given patient, doctor, date, time and duration. The method should only add the appointment when it does not overlap with any other appointment of the same doctor.
